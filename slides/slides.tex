\section{Problema e Objetivos}

\begin{frame}{Problema, Objetivo e Entregas}
    \begin{columns}[T]
        \column{0.48\textwidth}
        \textbf{Problema}
        \begin{itemize}
            \item Broker MQTT \'e ponto cr\'itico em sistemas IoT
            \item Mosquitto processa entradas de rede em C
            \item APIs POSIX externas bloqueiam a an\'alise pelo ESBMC
            \item Simular kernel e TCP/IP seria invi\'avel
        \end{itemize}
        \column{0.48\textwidth}
        \textbf{Objetivo e entregas}
        \begin{itemize}
            \item Abstrair a rede com modelos operacionais
            \item \textbf{9 fun\c{c}\~oes POSIX} modeladas
            \item \textbf{21 harnesses} simb\'olicos
            \item Avalia\c{c}\~ao em CVEs e no snapshot \mosqversion
            \item PR \#5388 e Security Report \#551
        \end{itemize}
    \end{columns}
    \vspace{0.6em}
    \begin{alertblock}{Problema de pesquisa}
        Como modelar a intera\c{c}\~ao com a rede para verificar vulnerabilidades
        reais do Eclipse Mosquitto com ESBMC?
    \end{alertblock}
\end{frame}

    \begin{tabular}{p{3.0cm}p{8.0cm}}
        \toprule
        API & Abstra\c{c}\~ao verificada \\
        \midrule
        \texttt{socket}, \texttt{accept} & descritores n\~ao determin\'isticos em faixa v\'alida \\
        \texttt{bind}, \texttt{listen}, \texttt{close} & retorno de sucesso sem simular o kernel \\
        \texttt{recv} & bytes e comprimento n\~ao determin\'isticos \\
        \texttt{send} & quantidade enviada n\~ao determin\'istica \\
        \texttt{select} & qualquer subconjunto de descritores prontos \\
        \texttt{poll} & eventos e erros n\~ao determin\'isticos \\
        \bottomrule
    \end{tabular}

    \vspace{0.5em}
    \begin{block}{Ideia central}
        \texttt{recv()} transforma a rede em entrada simb\'olica controlada pelo
        atacante, sem modelar TCP/IP.
    \end{block}
\end{frame}

\section{Achado de Seguran\c{c}a}

\begin{frame}{Falha no Snapshot \mosqversion}
    \begin{block}{\texttt{src/proxy\_v2.c} -- \texttt{read\_tlv\_ssl} (CWE-191)}
        Parser de sub-TLVs SSL no PROXY Protocol v2. O contador \texttt{len}
        representa os bytes restantes dentro da SSL TLV.
    \end{block}

    \vspace{0.3em}
    \textbf{Opera\c{c}\~ao vulner\'avel:}
    \begin{center}
        \texttt{len = (uint16\_t)(len - (3 + tlv\_len));}
    \end{center}
    sem verificar antes se \texttt{3 + tlv\_len <= len}.

    \vspace{0.5em}
    \textbf{Contraexemplo produzido pelo ESBMC:}
    \begin{tabular}{ll}
        \texttt{ssl\_tlv\_len = 256} & \texttt{len = 251} ap\'os o cabe\c{c}alho SSL \\
        \texttt{tlv\_len = 253} & passa no limite do buffer global \\
        \multicolumn{2}{l}{\texttt{3 + 253 = 256 > 251} $\Rightarrow$ \texttt{len = 65.531}} \\
    \end{tabular}

    \vspace{0.4em}
    \footnotesize Requer \texttt{proxy\_protocol true}. Reportado ao Eclipse em
    jun./2026 como Security Report \#551. \textbf{Resultado da QP4.}
\end{frame}

\begin{frame}{Por que o Harness Encontrou a Falha?}
    \begin{alertblock}{Checagens autom\'aticas e fuzzing}
        \small
        \begin{tabular}{p{4.2cm}p{6.8cm}}
            \texttt{--unsigned-overflow-check} & a opera\c{c}\~ao ocorre como \texttt{int} por promo\c{c}\~ao \\
            \texttt{--overflow-check} & o resultado negativo intermedi\'ario \'e v\'alido \\
            UBSan / libFuzzer & a convers\~ao \'e definida e n\~ao produz crash \\
        \end{tabular}
    \end{alertblock}

    \begin{block}{Propriedade de dom\'inio especificada no harness}
        \centering
        \texttt{\_\_ESBMC\_assert(3u + tlv\_len <= len, ...);}
    \end{block}

    \textbf{Corre\c{c}\~ao proposta:} rejeitar a TLV antes da subtra\c{c}\~ao se
    seu tamanho exceder o espa\c{c}o restante.

    \vspace{0.4em}
    \small \alert{FAILED} no c\'odigo original (3,71 s) \quad/\quad
    \textbf{SUCCESSFUL} com a guarda (2,17 s).

    \vspace{0.3em}
    \footnotesize Commit \texttt{23c918ee}: fuzzer adicionado ao subsistema em
    jan./2026; esta falha l\'ogica n\~ao gera um or\'aculo de crash.
\end{frame}

        \end{itemize}
    \end{columns}
    \vspace{0.5em}
    \small \textbf{Mitiga\c{c}\~ao:} pares vuln./fix, casos via \texttt{recv()} e
    comandos, fontes e logs versionados.

\section{Conclus\~ao}

\begin{frame}{Conclus\~ao}
    \begin{enumerate}
        \item Modelos operacionais tornam entradas de rede acess\'iveis ao ESBMC
        \item A metodologia reproduziu CVEs e distinguiu variantes corrigidas
        \item O custo foi baixo na maioria dos harnesses: \textbf{18/21 em $<6$ s}
        \item Uma asser\c{c}\~ao de dom\'inio revelou uma falha que n\~ao causa crash
    \end{enumerate}

    \vspace{0.8em}
    \textbf{\alert{Resposta \`a pergunta de pesquisa}}\\[0.2em]
    Modelos POSIX e harnesses simb\'olicos permitem verificar propriedades
    de brokers MQTT sem simular a pilha de rede.
\end{frame}
